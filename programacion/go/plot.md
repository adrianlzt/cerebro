func benchmarkTimePoints(data map[int]testing.BenchmarkResult) plotter.XYs {
  pts := plotter.XYs{}

  sortedMapIntString(data,
    func(k int, v testing.BenchmarkResult) {
      pts = append(pts, plotter.XY{X: float64(k), Y: float64(v.NsPerOp())})
    })
  return pts
}
